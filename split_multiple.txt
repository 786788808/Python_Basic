
@patch('builtins.open')
def test_sanction_config(mock_open_builtin):
    # 1. Mock env.json
    m1 = mock_open(read_data=json.dumps(MOCK_ENV_CONFIG))
    # 2. Mock WebS_US_Sanction.json
    m2 = mock_open(read_data=json.dumps(MOCK_PARAM_CONFIG))
    # 3. Mock log file
    m3 = mock_open()  # 日志文件不需要读取内容，只需要能写入
    
    # 根据文件名返回不同的 mock
    def side_effect(*args, **kwargs):
        if args[0].endswith('env.json'):
            return m1.return_value
        elif args[0].endswith('WebS_US_Sanction.json'):
            return m2.return_value
        else:  # 其他文件（比如日志）
            return m3.return_value
            
    mock_open_builtin.side_effect = side_effect

    config = SanctionConfig('dev', '20250704')
